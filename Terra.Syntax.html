{-# OPTIONS --cubical-compatible #-}

module Terra.Syntax where
open import Terra.Data.Level

private
  variable
    ℓ ℓ′ : Level

_$_ : {A : Set ℓ} {B : A -> Set ℓ′} -> ((a : A) -> B a) -> (a : A) -> B a
f $ x = f x

infixr 4 _$_

case_of_ : {A : Set ℓ} {B : Set ℓ′} -> A -> (A -> B) -> B
case v of f = f v

_፦_ : (T : Set ℓ) -> T -> T
a ፦ b = b

type-of : {A : Set ℓ} -> A -> Set ℓ
type-of {A = A} _ = A
